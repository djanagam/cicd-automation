    try:
        with open(json_file, 'r') as f:
            return json.load(f)
    except FileNotFoundError:
        logging.error(f"Error: The JSON file '{json_file}' was not found.")
        return None
    except json.JSONDecodeError:
        logging.error(f"Error: The JSON file '{json_file}' is not valid JSON.")
